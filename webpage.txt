  <script src="https://cdn.jsdelivr.net/npm/contentful@9.1.13/dist/contentful.browser.min.js"></script>
  <script>
    const client = contentful.createClient({
      space: 'YOUR_SPACE_ID',        // replace with your Contentful space ID
      accessToken: 'YOUR_ACCESS_TOKEN' // replace with your Contentful delivery API access token
    });

    // Fetch Menu Items
    client.getEntries({ content_type: 'menuItem' })
      .then(response => {
        const container = document.getElementById('menu-cards');
        response.items.forEach(item => {
          const card = document.createElement('div');
          card.className = 'card';
          card.innerHTML = `
            <img src="${item.fields.image.fields.file.url}" alt="${item.fields.name}">
            <h3>${item.fields.name}</h3>
            <p>${item.fields.description}</p>
            <p><strong>$${item.fields.price}</strong></p>
          `;
          container.appendChild(card);
        });
      });

    // Fetch Specials
    client.getEntries({ content_type: 'special' })
      .then(response => {
        const container = document.getElementById('specials-cards');
        response.items.forEach(item => {
          const card = document.createElement('div');
          card.className = 'card';
          card.innerHTML = `
            <img src="${item.fields.image.fields.file.url}" alt="${item.fields.title}">
            <h3>${item.fields.title}</h3>
            <p>${item.fields.description}</p>
          `;
          container.appendChild(card);
        });
      });

    // Fetch Events
    client.getEntries({ content_type: 'event' })
      .then(response => {
        const container = document.getElementById('events-cards');
        response.items.forEach(event => {
          const card = document.createElement('div');
          card.className = 'card';
          card.innerHTML = `
            <h3>${event.fields.title}</h3>
            <p>${event.fields.date}</p>
            <p>${event.fields.description}</p>
          `;
          container.appendChild(card);
        });
      });

    // Fetch Story Section
    client.getEntries({ content_type: 'story' })
      .then(response => {
        const story = response.items[0]?.fields.content || "Our story will be here soon!";
        document.getElementById('story-content').textContent = story;
      });

    // Fetch Gallery Images
    client.getEntries({ content_type: 'galleryImage' })
      .then(response => {
        const gallery = document.getElementById('gallery-images');
        response.items.forEach(img => {
          const imageEl = document.createElement('img');
          imageEl.src = img.fields.image.fields.file.url;
          imageEl.alt = img.fields.title || "Gallery Image";
          gallery.appendChild(imageEl);
        });
      });

  </script>
